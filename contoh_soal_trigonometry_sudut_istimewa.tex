\text{soal 2} \\

\text{soal 3} \\

$\frac{1}{4} \sqrt{2} + \frac{1}{4} \sqrt{6} =$ \\
$\frac{1}{4} (\sqrt{2} + \sqrt{6})$ \\

\\
$\frac{1}{4} * 3 - \frac{1}{4} =$ \\
\\
$\frac{3}{4} - \frac{1}{4} = \frac{2}{4}$
